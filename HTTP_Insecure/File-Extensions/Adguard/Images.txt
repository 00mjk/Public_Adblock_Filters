! Image File extensions: bmp, gif, jpeg, jpg,  png, psd, svg, tif, tiff, ico, djvu, djv
/^http://.+(/.+)?/.+\.(?:(?:[bB][mM][pP])|(?:[gG][iI][fF])|(?:[jJ][pP][eE]?[gG])|(?:[pP][nN][gG])|(?:[pP][sS][dD])|(?:[sS][vV][gG])|(?:[tT][iI][fF][fF]?)|(?:[iI][cC][oO])|(?:[dD][jJ][vV][uU]?))([#?].*)?$/$empty
