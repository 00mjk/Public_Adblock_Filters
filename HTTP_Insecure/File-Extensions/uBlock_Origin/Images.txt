! Due to uBlock Origin limitations, the rules below can only block URLs with the specific file extension only containing HTML content.
! Image File extensions: bmp, gif, jpeg, jpg, png, psd, svg, svgv, tif, tiff, ico, djvu, djv, webp, hief, hiec
/^http://.+(/.+)?/.+\.(?:(?:[bB][mM][pP])|(?:[gG][iI][fF])|(?:[jJ][pP][eE]?[gG])|(?:[pP][nN][gG])|(?:[pP][sS][dD])|(?:[sS][vV][gG][vV]?)|(?:[tT][iI][fF][fF]?)|(?:[iI][cC][oO])|(?:[dD][jJ][vV][uU]?)|(?:[wW][eE][bB][pP])|(?:[hH][iI][eE]([fF]|[cC]))([/#?].*)?$/$all
