! Use them at your own risk.

! Block downloads from HTTP domain names.
/^(?:http://[.0-9a-z-]+\.[a-z]{2,63}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw))$/$empty
! Block downloads from IPv4 addresses from HTTP traffic.
/^(?:http://(?:[0-9]{1,3}\.){3}[0-9]{1,3}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw))$/$empty
! Block downloads  from IPv6 addresses in standard notation.
/^(?:http://\[(?:[a-f0-9]{1,4}:){7}[a-f0-9]{1,4}\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw))$/$empty
