! This list does not block Javascript (.js) files.
/^http:\/\/.+(/.+){0,}/.+\.(?:(?:[eE][xX][eE])(?:[mM][sS][iI]))([/?].*)?$/$empty
/^https?:\/\/.+(/.+){0,}/.+\.(?:(?:[vV][bB]([sS]|[eE]))|(?:[jJ][sS][eE])|(?:[wW][sS]([fF]|[hH]))|(?:[sS][cC][rR])|(?:[pP][iI][fF])|(?:[rR][eE][gG])|(?:[jJ][aA][rR])|(?:[bB][aA][tT])|(?:[pP][sS][1])|(?:[pP][sS][2])|(?:[dD][lL][lL]))([/?].*)?$/$empty
