! This list does not block Javascript (.js) files.
! Block downloads from HTTP domain names.
/^(?:http://[.0-9a-z-]+\.[a-z]{2,63}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
! Block downloads from IPv4 addresses from HTTP traffic.
/^(?:http://(?:[0-9]{1,3}\.){3}[0-9]{1,3}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
! Block downloads  from IPv6 addresses in standard notation.
/^(?:http://\[(?:[a-f0-9]{1,4}:){7}[a-f0-9]{1,4}\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
! Block downloads from IPv6 addresses from HTTP traffic in compressed notation with at most 7 colons.
/^(?:http://\[(?:[a-f0-9]{0,4}:){1,7}[a-f0-9]{0,4}\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
! Block downloads from IPv6 addresses from HTTP traffic in compressed notation with at most 7 colons and 1 double colon.
/^(?:http://\[(([0-9a-f]{1,4}:){1,7}|:)((:[0-9a-f]{1,4}){1,7}|:)\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
! Block downloads from IPv6 addresses from HTTP traffic in compressed notation with at 8 colons.
/^(?:http://\[((?:[a-f0-9]{1,4}:){7}:|:(:[a-f0-9]{1,4}){7})\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:vbs|vbe|jse|wsf|wsh|hta|scr|pif|reg|jar|bat|ps1|exe|msi|dll))$/$empty
