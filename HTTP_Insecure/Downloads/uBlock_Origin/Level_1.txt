! Level 1: Executables + System
! Level 2: Executables + System + Archives
! Level 3: Executables + System + Archives + Documents
! Level 4: Executables + System + Archives + Documents + Images + Audio + Video + Generic 
! Executable File extensions: exe, apk, apks, app, appx, appxbundle, elf, bin, com, bat, gadget, msi, pif, msp, scr, hta, cpl, msc, jar
/^http://.+(/.+)?/.+\.(?:(?:[eE][xX][eE])|(?:[pP][iI][fF])|(?:[aA][pP][kK][sS]?)|(?:[aA][pP][pP][xX]?([bB][uU][nN][dD][lL][eE])?)|(?:[bB][iI][nN])|(?:[cC][oO][mM])|(?:[bB][aA][tT])|(?:[gG][aA][dD][gG][eE][tT])|(?:[mM][sS]([iI]|[pP]|[cC]))|(?:[sS][cC][rR])|(?:[hH][tT][aA])|(?:[jJ][aA][rR])|(?:[cC][pP][lL]))([#?].*)?$/$all
! Common Microsoft Windows scripts: cmd, vb, vbs, vbe, jse, ws, wsf, wsc, wsh, ps1, ps1xml, ps2, ps2xml, psc1, psc2, msh, msh1, msh2, mshxml, msh1xml, msh2xml
/^http://.+(/.+)?/.+\.(?:(?:[vV][bB]([sS]|[eE])?)|(?:[jJ][sS][eE])|(?:[wW][sS]([fF]|[hH]|[cC])?)|(?:[pP][sS][1]([xX][mM][lL])?)|(?:[pP][sS][2]([xX][mM][lL])?)|(?:[mM][sS][hH]([1]|[2])?([xX][mM][lL])?)|(?:[pP][sS][cC]([1]|[2])))([#?].*)?$/$all
! Other Microsoft Windows shortcuts, handles, libraries: dll, scr, inf, reg, chm, isp, lib, lnk, nsh, shb, sys, vxd
/^http://.+(/.+)?/.+\.(?:(?:[dD][lL][lL)|(?:[sS][cC][rR])|(?:[wW][iI][nN][fF])|(?:[rR][eE][gG])|(?:[cC][hH][mM])|(?:[iI][sS][pP])|(?:[lL][iI][bB])|(?:[lL][nN][kK])|(?:[nN][sS][hH])|(?:[sS][hH][bB])|(?:[sS][yY][sS])|(?:[vV][xX][dD]))([#?].*)?$/$all
