! Due to uBlock Origin limitations, the rules below can only block URLs with the specific file extension only containing HTML content.
! This list does not block Javascript (.js) files. File extensions: vb, vbs, vbe, jse, ws, wsc, wsf, wsh, scr, pif, reg, jar, bat, ps1, ps2, exe, msi, dll 
/^http:\/\/.+(/.+){0,}/.+\.(?:(?:[vV][bB]([sS]|[eE]))|(?:[jJ][sS][eE])|(?:[wW][sS]([fF]|[hH]|[cC])?)|(?:[sS][cC][rR])|(?:[pP][iI][fF])|(?:[rR][eE][gG])|(?:[jJ][aA][rR])|(?:[bB][aA][tT])|(?:[pP][sS][1])|(?:[pP][sS][2])|(?:[eE][xX][eE])(?:[mM][sS][iI])|(?:[dD][lL][lL]))([#?].*)?$/$all
