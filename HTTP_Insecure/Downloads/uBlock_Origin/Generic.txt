! Description: Generic file extensions. Block dangerous content that can be manipulated through HTTP connections and potentially harmful to devices. Intended for uBlock Origin.
! Use them at your own risk.
! ============================================================================
! Due to uBlock Origin limitations, the rules below can only block URLs with the specific file extension only containing HTML content.
/^http:\/\/(?:[!"#$%&'()*+,./0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+).*\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw)*$/$doc,popup
! Block HTTP domain names for generic file extensions.
/^(?:http://[.0-9A-Za-z-]+\.[A-Za-z]{2,63}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw)(?:\?[!"#$%&'()*+,./0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]*)?)*$/$all
! Block IPv4 addresses from HTTP traffic for generic file extensions.
/^(?:http://(?:[0-9]{1,3}\.){3}[0-9]{1,3}(?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw)(?:\?[!"#$%&'()*+,./0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]*)?)*$/$all
! Block IPv6 addresses from HTTP traffic in generic notation for Windows file extensions.
/^(?:http://\[(?:[A-Fa-f0-9]{1,4}:){7}[A-Fa-f0-9]{1,4}\](?:/[!"#$%&'()*+,.0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]+){0,}/[.0-9A-Z_a-z-]+\.(?:cnf|conf|cfg|log|asc|txt|bak|bk|bin|dat|dsk|raw)(?:\?[!"#$%&'()*+,./0-9:;<=>?@A-Z[\\\]_`a-z{|}~^-]*)?)*$/$all
